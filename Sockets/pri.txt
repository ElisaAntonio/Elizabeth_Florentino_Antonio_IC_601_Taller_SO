
modvideo macro v1    ;inicio del primer macro  
	mov ah,v1        ;mov en ah lo que el parametro v1 contiene
	int 10h          ;interrupcion 10h para el modo de video
	mov ah,00h       ;Establecer modo de Vídeo (40 por 25)
	int 10h          ;llamada a la interrupcion 10h 
endm                 ;termina el primero macro

cursor macro f,c     ;incia macro2 
	mov ah,02h       ;petición para colocar el cursor
	mov dh,f         ;linea donde situar el cursor
	mov dl,c         ;columna donde situar el cursor
	int 10h          ;llamada a la interrupcion 10h 
endm

escribir macro cad     ;inicio del macro escribir
	mov ah,09h         ;mueve a la parte alta del registro y llamar a funcion para desplejar una cadena de caracteres
	mov dx,pffset cad  ;leemos en dx lo que la variable cad tiene
	int 21h            ;llamada a la interrupción 21h
endm

end
